const button = document.querySelector(".btn");
const noteContainer = document.querySelector(".note-container");

//এখানে কি হলোঃ
      // ১. button এ ক্লিক করলে-> 
            // * html স্টাকচার হবে।


button.addEventListener("click", () => {
    let div = document.createElement("div");
    let p = document.createElement("p");
    let i = document.createElement("i");

    div.className = "noteBox";
    p.className = "inputBox";
    p.getAttribute("contentEditable", "true");
    i.className = "fa-solid fa-trash";

    noteContainer.appenChild(div);
    div.appenChild(p);
    div.appenChild(i);
    updateStorage()
});

// updateStorage() এ ডাটা আপডেট হবে।

function updateStorage () {
    localStorage.setItem("note", noteContainer.innerHtml)
};

// saveStorage() এ ডাটা save হবে।
function saveStorage () {
    noteContainer.innerHtml = localStorage.getItem("note")
}
saveStorage();


//১. noteContainer এ ক্লিক করলে event রান হবে।
        // * যদি noteContainer এর ভেতরে <i> তে ক্লিক করা হয় -> noteContainer পুরোটাই রিমুভ হবে না।
        // * update হবে ডাটা

        
noteContainer.addEventListener("click", (e) => {
    if(e.target.tagName === "I"){
        e.target.parentElement.remove();
        updateStorage()
    }
})
