let currentQuestion = 0;
skipBtn.addEventListener("click", showQuizQuestion);
function showQuizQuestion(){
    let contentCard = card.length -1;

    card.forEach(card => card.classList.remove('active'));
    if(currentQuestion < contentCard){
        currentQuestion++;
    }

    card[currentQuestion].classList.add('active');

    if(currentQuestion === contentCard){
        skipBtn.style.display = "none";
    }
    resultScore.innerHtml = count;
};

let count = 0;

    input.addEventListener("change", (e) => {
