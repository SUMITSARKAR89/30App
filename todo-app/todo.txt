function addBtn (){
    const inputValue = input.value;
    if(inputValue === ""){
        alert("please add task");
    }else {
        let liTag = document.creatElement("li");
        liTag.innerHtml = inputValue;
        listBox.appenchild(liTag)

        let spanTag = document.creatElement("span");
        spanTag.innerHtml = ".\icon.png";
        liTag.appenchild(spanTag)
    };
    inputValue = '';
    saveItem();
};


//এখানে কি করা হলোঃ
১. <ul> এর ভেতরে <li> কে ক্লিক করলে একটা event পাস হবে
২. cleckTarget নামে একটা value নেয়া হবে।
৩. যদি tagName <li> তে ক্লিক করা হয় তাহলে toggle এর মাধ্যমে .checked ক্লাস কাজ করবে।
৪. যদি tagName <span> তে ক্লিক করা হয় তাহলে <li> parentElemet টা remove হয়ে যাবে।



listBox.addEventListener("click", (e)=> {
    const clickTarget = e.target;
    if(clickTarget.tagName === "LI){
        clickTarget.classList.toggle("checked");
        saveItem();
    }else(clickTarget.tagName === "SPAN"){
        clickTarget.parentElemet.remove();
        saveItem();
    }

});

//এখানে কি করা হলোঃ
১. data গুলো যেন set করতে
        * saveItem() fanction এর মাধ্যমে local storage এ <li> এর ডাটাগুলো set করবো।
২. save করা data গুলো যেন পাওয়া যায়
        * showItem() function এর মাধ্যমে local storage এ <li> এর ডাটাগুলো get করবো ।      

function saveItem (){
    localStorage.setItem("demo", listBox.innerHtml)
}

function showItem (){
    listBox.innerHtml = localStorage.getItem("demo")
}
showItem();
